How LogsDX Works

  1. Schema Definition: Themes define patterns using matchWords, matchPatterns, etc.
  2. Tokenization: Log lines are broken into tokens based on schema patterns
  3. Theme Application: Styles are applied to matching tokens
  4. Multi-format Rendering:
    • Terminal: ANSI escape codes
    • Browser: HTML with CSS styles or classes

Result: Same visual appearance across environments!

        function mockProcessLine(line, theme, format) {
            // This is a simplified mock - real LogsDX does much more sophisticated pattern matching
            const themeConfig = themes[theme];
            if (!themeConfig) return line;

            let processed = line;
            
            // Simple pattern matching for demo
            if (line.includes('[INFO]')) {
                if (format === 'html') {
                    processed = processed.replace('[INFO]', `[INFO]`);
                }
            }
            if (line.includes('[ERROR]')) {
                if (format === 'html') {
                    processed = processed.replace('[ERROR]', `[ERROR]`);
                }
            }
            if (line.includes('[WARN]')) {
                if (format === 'html') {
                    processed = processed.replace('[WARN]', `[WARN]`);
                }
            }
            if (line.includes('[DEBUG]')) {
                if (format === 'html') {
                    processed = processed.replace('[DEBUG]', `[DEBUG]`);
                }
            }

            // Style URLs
            processed = processed.replace(/(\/[a-zA-Z0-9\/\-_]+)/g, format === 'html' ? 
                '$1' : '$1');
            
            // Style status codes
            processed = processed.replace(/\b([1-5][0-9]{2})\b/g, format === 'html' ? 
                '$1' : '$1');

            return processed;
        }

        function updateDemo() {
            const selectedTheme = document.getElementById('themeSelect').value;
            const terminalOutput = document.getElementById('terminalOutput');
            const browserOutput = document.getElementById('browserOutput');

            // Terminal output (mock ANSI by showing raw format)

            // Browser output (HTML with styles)

                `

`
            ).join('');
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
